\documentclass[tikz,margin=.5cm]{standalone}
\usepackage{ortho-route}

\begin{document}

% N -> E 
\def\ne{
\portho (a.north) to (b.east);
\portho (a.north) to (c.east);
\portho (a.north) to (d.east);
\portho (a.north) to (e.east);
}

% N -> N
\def\nn{
\portho (a.north) to (b.north);
\portho (a.north) to (c.north);
\portho (a.north) to (d.north);
\portho (a.north) to (e.north);
}

% test rotation logic (equivalent to above but rotated)
\def\ee{
\portho (a.east) to (b.east);
\portho (a.east) to (c.east);
\portho (a.east) to (d.east);
\portho (a.east) to (e.east);
}

\def\ss{
\portho (a.south) to (b.south);
\portho (a.south) to (c.south);
\portho (a.south) to (d.south);
\portho (a.south) to (e.south);
}

\def\ww{
\portho (a.west) to (b.west);
\portho (a.west) to (c.west);
\portho (a.west) to (d.west);
\portho (a.west) to (e.west);
}

% N -> W
\def\nw{
\portho (a.north) to (b.west);
\portho (a.north) to (c.west);
\portho (a.north) to (d.west);
\portho (a.north) to (e.west);
}

% N -> S (will generally require more node separation since 2 times in direction
\def\ns{
\portho (a.north) to (b.south);
\portho (a.north) to (c.south);
\portho (a.north) to (d.south); % ERROR case without S-curve
\portho (a.north) to (e.south); % ERROR case without S-curve
}

\def\sn{ % another test of rotation logic
\portho (a.south) to (b.north);
\portho (a.south) to (c.north);
\portho (a.south) to (d.north); % ERROR case without S-curve
\portho (a.south) to (e.north); % ERROR case without S-curve
}

\def\limiting{
% limiting cases
\node[above=1cm] (an) at (a) {};
\node[below=1cm] (as) at (a) {};
\node[right=1cm] (ae) at (a) {};
\node[left=1cm] (aw) at (a) {};
\portho (a.north) to (an.south);
\portho (a.north) to (ae.west);
\portho (a.north) to (as.north);
\portho (a.north) to (aw.east);
}

\node (d) at (-1,-1) {};

\end{document}
